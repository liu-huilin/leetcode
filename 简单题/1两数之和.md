### 1、暴力求解

```c++
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        int len = nums.size();
        for(int i=0; i<len-1; i++)
        {
            for(int j=i+1; j<len; j++)
            {
                if((nums[i]+nums[j]) == target)
                {
                   return {i, j};
                }
            }
        }
        return {0, 0};
    }
};
```

```c++
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map <int, int> has;
        int n = nums.size();
        for(int i = n - 1; i >= 0; --i){
            if(has.find(target - nums[i]) != has.end()){
                return {has[target - nums[i]], i};
            }
            has[ nums[i] ] = i;
        }
        return {};
    }
};
```
